import pandas as pd
import scipy.stats as stats

# 读取数据
data = pd.read_csv("deduped_table.csv")

# 分成两组
has_view_data = data[data["has_view_to_waterfront"] == True]["price"]
no_view_data = data[data["has_view_to_waterfront"] == False]["price"]

# 计算两组样本的均值和标准差
has_view_mean = has_view_data.mean()
has_view_std = has_view_data.std()

no_view_mean = no_view_data.mean()
no_view_std = no_view_data.std()

# 计算 t 统计量和 p 值
t_stat, p_val = stats.ttest_ind(has_view_data, no_view_data, equal_var=False)

# 输出结果
print("房子有海景的均价为：{:.2f}，标准差为：{:.2f}".format(has_view_mean, has_view_std))
print("房子无海景的均价为：{:.2f}，标准差为：{:.2f}".format(no_view_mean, no_view_std))
print("t 统计量为：{:.2f}，p 值为：{:.2f}".format(t_stat, p_val))
